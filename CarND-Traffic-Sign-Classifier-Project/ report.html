Step 1: Dataset Summary & Exploration¶

The pickled data is a dictionary with 4 key/value pairs:

  • 'features' is a 4D array containing raw pixel data of the traffic sign images, (num examples, width, height, channels).

def counts(y, Titel):
    
    labels = pd.DataFrame(y)
    dup_classes = labels.pivot_table(index=[0], aggfunc='size')
    dup_classes = dup_classes.tolist()

    classes = np.array(list(range(0,43)))
    classes = classes.T 
    final = pd.DataFrame(data = [classes,dup_classes])
    final = final.transpose()
    avg = int(final[1].mean())
    
    plt.figure(figsize=(15,5))
    plt.xticks(np.arange(min(final.index), max(final.index)+1, 1.0))
    plt.ylabel ('Count')
    plt.xlabel ('Class')
    plt.bar(final.index,final[1], label=Titel)
    plt.legend (bbox_to_anchor=(1, 1), loc="upper right", borderaxespad=0.)
    print ("Average number of images per class for",Titel + "= ", avg) 
    return avg,final

def grayscale_normalize (data):
    
    # Convert the images to grayscale

                data = np.append(data, [modi], axis=0)
                counter += 1
                if counter > needed:
                    break
            print ("Done augmenting - number of instances is now = ", num+counter)         
    return data, y

from tensorflow.contrib.layers import flatten

def LeNet(x):    
    # Arguments used for tf.truncated_normal, randomly defines variables for the weights and biases for each layer
    mu = 0
    sigma = 0.1
    
    # TODO: Layer 1: Convolutional. Input = 32x32x1. Output = 28x28x6.
    conv1_in_height = 32
    conv1_in_width = 32
    conv1_in_depth = 1
    
    conv1_out_height = 28 # Filter hight = 5
    conv1_out_width = 28 # Filter width = 5
    conv1_out_depth = 6 # Filter depth = 6
    
    conv1_strides = [1,1,1,1]
    
    conv1_out_height = math.ceil(float(conv1_in_height - conv1_out_height + 1) / float(conv1_strides[1]))
    conv1_out_width  = math.ceil(float(conv1_in_width - conv1_out_width + 1) / float(conv1_strides[2]))
    conv1_w = tf.Variable(tf.truncated_normal([conv1_out_height,conv1_out_width,conv1_in_depth,conv1_out_depth], mu, sigma))
    conv1_b = tf.Variable(tf.zeros(conv1_out_depth))
    conv1 = tf.nn.conv2d(x, conv1_w, conv1_strides, 'VALID') + conv1_b

    # TODO: Activation.
    conv1 = tf.nn.relu(conv1)

    # TODO: Pooling. Input = 28x28x6. Output = 14x14x6.
    conv1 = tf.nn.max_pool(conv1, [1,2,2,1], [1,2,2,1], 'VALID')
    

    # TODO: Layer 2: Convolutional. Output = 10x10x16.
    conv2_in_height = 14
    conv2_in_width = 14
    conv2_in_depth = 6
   
    conv2_out_height = 10 # Filter hight = 5
    conv2_out_width = 10 # Filter width = 5
    conv2_out_depth = 16 # Filter depth = 16
   
    conv2_strides = [1,1,1,1]
   
    conv2_out_height = math.ceil(float(conv2_in_height - conv2_out_height + 1) / float(conv2_strides[1]))
    conv2_out_width  = math.ceil(float(conv2_in_width - conv2_out_width + 1) / float(conv2_strides[2]))
    conv2_w = tf.Variable(tf.truncated_normal([conv2_out_height,conv2_out_width,conv2_in_depth,conv2_out_depth], mu, sigma))
    conv2_b = tf.Variable(tf.zeros(conv2_out_depth))
    conv2 = tf.nn.conv2d(conv1, conv2_w, conv2_strides, 'VALID') + conv2_b
    
    # TODO: Activation.
    conv2 = tf.nn.relu(conv2)

    # TODO: Pooling. Input = 10x10x16. Output = 5x5x16.
    conv2 = tf.nn.max_pool(conv2, [1,2,2,1], [1,2,2,1], 'VALID')
    conv2 = tf.nn.dropout(conv2, keep_prob)
    

    # TODO: Flatten. Input = 5x5x16. Output = 400.
    dlayer1 = flatten(conv2)
    
    # TODO: Layer 3: Fully Connected. Input = 400. Output = 120.
    dlayer1_w = tf.Variable(tf.truncated_normal([400,120], mu, sigma))
    dlayer1_b = tf.Variable(tf.zeros(120))
    #dlayer1 = tf.reshape(dlayer1, [-1, dlayer1_w.get_shape().as_list()[0]])
    dlayer1 = tf.add(tf.matmul(dlayer1, dlayer1_w),dlayer1_b)
    
    
    # TODO: Activation.
    dlayer1 = tf.nn.relu(dlayer1)
    dlayer1 = tf.nn.dropout(dlayer1, keep_prob)

    # TODO: Layer 4: Fully Connected. Input = 120. Output = 84.
    dlayer2_w = tf.Variable(tf.truncated_normal([120,84], mu, sigma))
    dlayer2_b = tf.Variable(tf.zeros(84))
    #dlayer2 = tf.reshape(dlayer1, [-1, dlayer2_w.get_shape().as_list()[0]])
    dlayer2 = tf.add(tf.matmul(dlayer1, dlayer2_w), dlayer2_b)
    
    
    # TODO: Activation.
    dlayer2 = tf.nn.relu(dlayer2)
    dlayer2 = tf.nn.dropout(dlayer2, keep_prob)

    # TODO: Layer 5: Fully Connected. Input = 84. Output = 43.
    logits_w = tf.Variable(tf.truncated_normal([84,n_classes], mu, sigma))
    logits_b = tf.Variable(tf.zeros(n_classes))
    #logits = tf.reshape(dlayer2, [-1, logits_w.get_shape().as_list()[0]])
    #logits = tf.nn.dropout(logits, keep_prob)
    logits = tf.add(tf.matmul(dlayer2, logits_w),logits_b)
    
    
    return logits

def evaluate(X_data, y_data):
    num_examples = len(X_data)
    total_accuracy = 0
    sess = tf.get_default_session()
    for offset in range(0, num_examples, BATCH_SIZE):
        batch_x, batch_y = X_data[offset:offset+BATCH_SIZE], y_data[offset:offset+BATCH_SIZE]
        accuracy = sess.run(accuracy_operation, feed_dict={x: batch_x, y: batch_y, keep_prob : 1.0})
        total_accuracy += (accuracy * len(batch_x))
    return total_accuracy / num_examples

Output Top 5 Softmax Probabilities For Each Image Found on the Web¶

For each of the new images, print out the model's softmax probabilities to show the certainty of the model's predictions (limit the output to the top 5 probabilities for each image). tf.nn.top_k could prove helpful here.

The example below demonstrates how tf.nn.top_k can be used to find the top k predictions for each image.

tf.nn.top_k will return the values and indices (class ids) of the top k predictions. So if k=3, for each sign, it'll return the 3 largest probabilities (out of a possible 43) and the correspoding class ids.

Take this numpy array as an example. The values in the array represent predictions. The array contains softmax probabilities for five candidate images with six possible classes. tf.nn.top_k is used to choose the three classes with the highest probability:
